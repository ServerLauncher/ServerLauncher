cmake_minimum_required(VERSION 3.19)

project(MSC LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(CMAKE_PREFIX_PATH "C:/msys64/ucrt64")

find_package(Qt6 REQUIRED COMPONENTS Core Widgets Network)

qt_standard_project_setup()

file(GLOB_RECURSE SOURCES src/*.cpp)
file(GLOB_RECURSE HEADERS include/*.hpp)
file(GLOB_RECURSE UIS src/*.ui)

qt_add_executable(MSC ${SOURCES} ${HEADERS} ${UIS})

if (WIN32) 
    set_target_properties(MSC PROPERTIES
        WIN32_EXECUTABLE TRUE
    )

    target_sources(MSC PRIVATE ${CMAKE_SOURCE_DIR}/resources/serverlauncher.rc)
endif()

set_target_properties(MSC PROPERTIES
    AUTOUIC_SEARCH_PATHS "${CMAKE_SOURCE_DIR}/src"
)

target_include_directories(MSC PRIVATE 
    include 
    ${CMAKE_CURRENT_BINARY_DIR}
)

target_link_libraries(MSC PRIVATE 
    Qt6::Core 
    Qt6::Widgets
    Qt6::Network
)

find_package(cpr CONFIG REQUIRED)
target_link_libraries(MSC PRIVATE cpr::cpr)

find_package(nlohmann_json REQUIRED)
target_link_libraries(MSC PRIVATE nlohmann_json::nlohmann_json)

find_package(spdlog REQUIRED)
target_link_libraries(MSC PRIVATE spdlog::spdlog)
