cmake_minimum_required(VERSION 3.10)

project(MSC)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

file(GLOB_RECURSE SOURCES src/*.cpp)

add_executable(MSC ${SOURCES})

target_include_directories(MSC PRIVATE include include/downloaders)

find_package(cpr CONFIG REQUIRED)
target_link_libraries(MSC PRIVATE cpr::cpr)

find_package(spdlog REQUIRED)
target_link_libraries(MSC PRIVATE spdlog::spdlog)
